   369→// ────────────────────────────────────────────────────────────
   370→// Bulk actions component

   372→function BulkActions({ tenantIds, providerTypes, onDone }: { tenantIds: string[]; providerTypes: ProviderTypeMeta[]; onDone: () => void }) {
   373→  const [createOpen, setCreateOpen] = React.useState(false);
   374→  const [busy, setBusy] = React.useState<string | null>(null);
   375→
   376→  const runBulk = async (action: string, extra: any = {}) => {
   377→    setBusy(action);
   378→    try {
   379→      const res = await api<{ successCount: number; failCount: number }>("/api/ai-providers/bulk", {
   380→        method: "POST",
   381→        body: JSON.stringify({ action, tenantIds, ...extra }),
   382→      });
   383→      if (res.failCount === 0) {
   384→        toast.success(`عملیات با موفقیت برای ${toFa(res.successCount)} کسب‌وکار انجام شد.`);
   385→      } else {
   386→        toast.warning(`${toFa(res.successCount)} موفق، ${toFa(res.failCount)} ناموفق.`);
   387→      }
   388→      onDone();
   389→    } catch (e: any) {
   390→      toast.error(e.message);
   391→    } finally {
   392→      setBusy(null);
   393→    }
   394→  };
   395→
   396→  return (
   397→    <div className="grid sm:grid-cols-2 gap-3">
   398→      {/* Create provider for all */}
   399→      <Dialog open={createOpen} onOpenChange={setCreateOpen}>
   400→        <Card className="p-4">
   401→          <div className="flex items-center gap-2 mb-2">
   402→            <Plus className="size-4 text-primary" />
   403→            <h4 className="font-medium text-sm">ایجاد ارائه‌دهنده</h4>
   404→          </div>
   405→          <p className="text-xs text-muted-foreground mb-3 leading-5">
   406→            یک ارائه‌دهنده جدید (با همان کلید و مدل) برای همه {toFa(tenantIds.length)} کسب‌وکار ایجاد کنید.
   407→          </p>
   408→          <Button className="w-full gap-1.5" onClick={() => setCreateOpen(true)} disabled={!!busy}>
   409→            <Plus className="size-4" /> ایجاد دسته‌جمعی
   410→          </Button>
   411→        </Card>
   412→        <ProviderFormDialog
   413→          open={createOpen} onOpenChange={setCreateOpen}
   414→          providerTypes={providerTypes}
   415→          title={`ایجاد ارائه‌دهنده برای ${toFa(tenantIds.length)} کسب‌وکار`}
   416→          submitLabel="ایجاد برای همه"
   417→          onSubmit={async (data) => {
   418→            await runBulk("create", data);
   419→            setCreateOpen(false);
   420→          }}
   421→        />
   422→      </Dialog>
   423→
   424→      {/* Deactivate all */}
   425→      <Card className="p-4">
   426→        <div className="flex items-center gap-2 mb-2">
   427→          <Power className="size-4 text-amber-600" />
   428→          <h4 className="font-medium text-sm">غیرفعال‌سازی همه</h4>
   429→        </div>
   430→        <p className="text-xs text-muted-foreground mb-3 leading-5">
   431→          همه {toFa(tenantIds.length)} کسب‌وکار را به موتور پیش‌فرض پلتفرم (Z.ai) بازگردانید.
   432→        </p>
   433→        <Button variant="outline" className="w-full gap-1.5" onClick={() => runBulk("deactivate")} disabled={!!busy}>
   434→          {busy === "deactivate" ? <Loader2 className="size-4 animate-spin" /> : <Power className="size-4" />}
   435→          بازگشت همه به پیش‌فرض
   436→        </Button>
   437→      </Card>
   438→    </div>
   439→  );
   440→}

   442→// ────────────────────────────────────────────────────────────
   443→// Reusable provider form dialog (used for single + bulk create)
   444→// ────────────────────────────────────────────────────────────
   445→function ProviderFormDialog({
   446→  open, onOpenChange, providerTypes, title, onSubmit, submitLabel,
   447→}: {
   448→  open: boolean; onOpenChange: (o: boolean) => void;
   449→  providerTypes: ProviderTypeMeta[]; title: string;
   450→  onSubmit: (data: { name: string; type: string; apiKey: string; baseUrl: string; model: string }) => Promise<void>;
   451→  submitLabel?: string;
   452→}) {
   453→  const [type, setType] = React.useState("openai");
   454→  const [name, setName] = React.useState("");
   455→  const [apiKey, setApiKey] = React.useState("");
   456→  const [baseUrl, setBaseUrl] = React.useState("");
   457→  const [model, setModel] = React.useState("");
   458→  const [submitting, setSubmitting] = React.useState(false);
   459→  const [testing, setTesting] = React.useState(false);
   460→
   461→  const pt = providerTypes.find((p) => p.code === type);
   462→
   463→  React.useEffect(() => {
   464→    if (pt) { if (!model) setModel(pt.defaultModel); if (!name) setName(pt.label); }
   465→  }, [type]);
   466→
   467→  const reset = () => { setType("openai"); setName(""); setApiKey(""); setBaseUrl(""); setModel(""); };
   468→
   469→  const submit = async () => {
   470→    if (!name.trim()) { toast.error("نام الزامی است"); return; }
   471→    if (pt?.needsKey && !apiKey.trim()) { toast.error("کلید API الزامی است"); return; }
   472→    if (pt?.needsBaseUrl && !baseUrl.trim()) { toast.error("Base URL الزامی است"); return; }
   473→    setSubmitting(true);
   474→    try {
   475→      await onSubmit({ name: name.trim(), type, apiKey: apiKey.trim(), baseUrl: baseUrl.trim(), model: model.trim() || pt?.defaultModel || "" });
   476→      reset();
   477→    } catch (e: any) { toast.error(e.message); }
   478→    finally { setSubmitting(false); }
   479→  };
   480→
   481→  const testInline = async () => {
   482→    if (pt?.needsKey && !apiKey.trim()) { toast.error("ابتدا کلید API را وارد کنید"); return; }
   483→    setTesting(true);
   484→    try {
   485→      const res = await api<{ ok: boolean; reply: string; error?: string }>("/api/ai-providers/test", {
   486→        method: "POST", body: JSON.stringify({ type, apiKey, baseUrl, model: model || pt?.defaultModel }),
   487→      });
   488→      if (res.ok) toast.success(`موفق! ${res.reply.slice(0, 50)}`);
   489→      else toast.error(`ناموفق: ${res.error}`);
   490→    } catch (e: any) { toast.error(e.message); }
   491→    finally { setTesting(false); }
   492→  };
   493→
   494→  return (
   495→    <DialogContent className="max-w-lg" open={open} onOpenChange={(o) => { onOpenChange(o); if (!o) reset(); }}>
   496→      <DialogHeader>
   497→        <DialogTitle className="flex items-center gap-2"><Plus className="size-4" /> {title}</DialogTitle>
   498→        <DialogDescription>پیکربندی مدل یا API اختصاصی</DialogDescription>
   499→      </DialogHeader>
   500→      <div className="space-y-3 max-h-[60vh] overflow-y-auto scroll-area pl-1">
   501→        <div className="space-y-1.5">
   502→          <Label>نوع ارائه‌دهنده</Label>
   503→          <Select value={type} onValueChange={setType}>
   504→            <SelectTrigger className="w-full"><SelectValue /></SelectTrigger>
   505→            <SelectContent>
   506→              {providerTypes.map((p) => <SelectItem key={p.code} value={p.code}>{p.label}</SelectItem>)}
   507→            </SelectContent>
   508→          </Select>
   509→          {pt && <p className="text-xs text-muted-foreground leading-5">{pt.desc}</p>}
   510→        </div>
   511→        <div className="space-y-1.5">
   512→          <Label>نام (نمایشی)</Label>
   513→          <Input value={name} onChange={(e) => setName(e.target.value)} placeholder="مثلاً: OpenAI تولیدی" />
   514→        </div>
   515→        {pt?.needsKey && (
   516→          <div className="space-y-1.5">
   517→            <Label className="flex items-center gap-1.5"><Key className="size-3" /> کلید API</Label>
   518→            <Input value={apiKey} onChange={(e) => setApiKey(e.target.value)} type="password" dir="ltr" placeholder="sk-..." className="font-mono text-left" />
   519→          </div>
   520→        )}
   521→        {pt?.needsBaseUrl && (
   522→          <div className="space-y-1.5">
   523→            <Label className="flex items-center gap-1.5"><Globe className="size-3" /> Base URL</Label>
   524→            <Input value={baseUrl} onChange={(e) => setBaseUrl(e.target.value)} dir="ltr" placeholder="https://your-endpoint/v1" className="font-mono text-left" />
   525→          </div>
   526→        )}
   527→        <div className="space-y-1.5">
   528→          <Label className="flex items-center gap-1.5"><Cpu className="size-3" /> شناسه مدل</Label>
   529→          <Input value={model} onChange={(e) => setModel(e.target.value)} dir="ltr" placeholder={pt?.defaultModel} className="font-mono text-left" />
   530→          {pt && <p className="text-[10px] text-muted-foreground">پیشنهاد: {pt.defaultModel}</p>}
   531→        </div>
   532→      </div>
   533→      <DialogFooter className="gap-2 sm:gap-2">
   534→        <Button variant="outline" onClick={testInline} disabled={testing} className="gap-1.5">
   535→          {testing ? <Loader2 className="size-4 animate-spin" /> : <Zap className="size-4" />}
   536→          تست اتصال
   537→        </Button>
   538→        <Button onClick={submit} disabled={submitting} className="gap-1.5">
   539→          {submitting ? <Loader2 className="size-4 animate-spin" /> : <Plus className="size-4" />}
   540→          {submitLabel || "افزودن"}
   541→        </Button>
   542→      </DialogFooter>
   543→    </DialogContent>
   544→  );
   545→}
